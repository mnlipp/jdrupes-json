019package org.jdrupes.json;
020
021/**
022 * 
023 */
024@SuppressWarnings("serial")
025public class JsonDecodeException extends Exception {
026
027        public JsonDecodeException() {
028                super();
029        }
030
031        public JsonDecodeException(String message, Throwable cause,
032                boolean enableSuppression, boolean writableStackTrace) {
033                super(message, cause, enableSuppression, writableStackTrace);
034        }
035
036        public JsonDecodeException(String message, Throwable cause) {
037                super(message, cause);
038        }
039
040        public JsonDecodeException(String message) {
041                super(message);
042        }
043
044        public JsonDecodeException(Throwable cause) {
045                super(cause);
046        }
047
048}
