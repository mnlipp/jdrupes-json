021import java.beans.BeanInfo;
022import java.beans.IntrospectionException;
023import java.beans.Introspector;
024import java.beans.PropertyDescriptor;
025import java.beans.PropertyEditor;

029import java.io.Flushable;
030import java.io.IOException;

043import javax.json.Json;
044import javax.json.stream.JsonGenerator;

046/**
047 * Encoder for converting a Java object graph to JSON. Objects may be arrays,
048 * collections, maps and JavaBeans.
049 * 
050 * Arrays and collections are converted to JSON arrays. The
051 * type information is lost. Maps and JavaBeans are converted
052 * to JSON objects.
053 * 
054 * The generated JSON objects can have an additional key/value pair with
055 * key "class" and a class name. The class information is generated
056 * only when it is needed, i.e. if it cannot be derived from the containing
057 * object.
058 * 
059 * Given the following classes:
060 *
061 * ```java
062 * public static class Person {
063 *
064 *     private String name;
065 *     private int age;
066 *     private PhoneNumber[] numbers;
067 *
068 *     public String getName() {
069 *         return name;
070 *     }
071 *     
072 *     public void setName(String name) {
073 *         this.name = name;
074 *     }
075 *     
076 *     public int getAge() {
077 *         return age;
078 *     }
079 *     
080 *     public void setAge(int age) {
081 *         this.age = age;
082 *     }
083 *     
084 *     public PhoneNumber[] getNumbers() {
085 *         return numbers;
086 *     }
087 *     
088 *     public void setNumbers(PhoneNumber[] numbers) {
089 *         this.numbers = numbers;
090 *     }
091 * }
092 *
093 * public static class PhoneNumber {
094 *     private String name;
095 *     private String number;
096 *
097 *     public PhoneNumber() {
098 *     }
099 *     
100 *     public String getName() {
101 *         return name;
102 *     }
103 *     
104 *     public void setName(String name) {
105 *         this.name = name;
106 *     }
107 *     
108 *     public String getNumber() {
109 *         return number;
110 *     }
111 *     
112 *     public void setNumber(String number) {
113 *         this.number = number;
114 *     }
115 * }
116 *
117 * public static class SpecialNumber extends PhoneNumber {
118 * }
119 * ```
120 * 
121 * A serialization result may look like this:
122 * 
123 * ```json
124 * {
125 *     "age": 42,
126 *     "name": "Simon Sample",
127 *     "numbers": [
128 *         {
129 *             "name": "Home",
130 *             "number": "06751 51 56 57"
131 *         },
132 *         {
133 *             "class": "test.json.SpecialNumber",
134 *             "name": "Work",
135 *             "number": "030 77 35 44"
136 *         }
137 *     ]
138 * } 
139 * ```
140 * 
141 * 
142 */
143public class JsonBeanEncoder extends JsonCoder 
144        implements Flushable, Closeable {
145
146        private static final Set<String> EXCLUDED_DEFAULT = new HashSet<>();
147
148        static {
149                // See https://issues.apache.org/jira/browse/GROOVY-8284
150                EXCLUDED_DEFAULT.add("groovy.lang.MetaClass");
151        }
152
153        private Map<Class<?>, String> aliases = new HashMap<>();
154        private Set<String> excluded = EXCLUDED_DEFAULT;
155        private JsonGenerator gen;
156        private StringWriter writer = null;
157
158        @Override
159        public JsonBeanEncoder addAlias(Class<?> clazz, String alias) {
160                aliases.put(clazz, alias);
161                return this;
162        }
163
164        /**
165         * Add a type to excude from encoding, usually because it cannot
166         * be converted to JSON. Properties of such types should be
167         * marked as {@link Transient}. However, sometimes base types
168         * don't follow the rules.
169         * 
170         * @param className
171         * @return the encoder for easy chaining
172         */
173        public JsonBeanEncoder addExcluded(String className) {
174                if (excluded == EXCLUDED_DEFAULT) {
175                        excluded = new HashSet<>(EXCLUDED_DEFAULT);
176                }
177                excluded.add(className);
178                return this;
179        }
180
181        /**
182         * Create a new encoder using a default {@link JsonGenerator}. 
183         * 
184         * @param out the sink
185         * @return the encoder
186         */
187        public static JsonBeanEncoder create(Writer out) {
188                return new JsonBeanEncoder(Json.createGenerator(out));
189        }
190
191        /**
192         * Create a new encoder using a default {@link JsonGenerator}
193         * that writes to an internally created {@link StringWriter}. 
194         * The result can be obtained by invoking {@link #toJson()}.
195         * 
196         * @return the encoder
197         */
198        public static JsonBeanEncoder create() {
199                return new JsonBeanEncoder();
200        }
201
202        /**
203         * Create a new encoder using the given {@link JsonGenerator}. 
204         * 
205         * @param generator the generator
206         * @return the encoder
207         */
208        public static JsonBeanEncoder create(JsonGenerator generator) {
209                return new JsonBeanEncoder(generator);
210        }
211
212        private JsonBeanEncoder() {
213                writer = new StringWriter();
214                gen = Json.createGenerator(writer);
215        }
216
217        private JsonBeanEncoder(JsonGenerator generator) {
218                gen = generator;
219        }
220
221        @Override
222        public void flush() throws IOException {
223                gen.flush();
224        }
225
226        @Override
227        public void close() throws IOException {
228                gen.close();
229        }
230
231        /**
232         * Returns the text written to the output. Can only be used
233         * if the encoder has been created with {@link #JsonBeanEncoder()}.
234         * 
235         * @return the result
236         */
237        public String toJson() {
238                if (writer == null) {
239                        throw new IllegalStateException(
240                                        "JsonBeanEncoder has been created without a known writer.");
241                }
242                gen.flush();
243                return writer.toString();
244        }
245        
246        public JsonBeanEncoder writeArray(Object... items) {
247                doWriteObject(items, items.getClass());
248                return this;
249        }
250        
251        public JsonBeanEncoder writeObject(Object obj) {
252                doWriteObject(obj, obj.getClass());
253                return this;
254        }
255        
256        private void doWriteObject(Object obj, Class<?> expectedType) {
257                if (obj == null) {
258                        gen.writeNull();
259                        return;
260                }
261                if (obj instanceof Boolean) {
262                        gen.write((Boolean)obj);
263                        return;
264                } 
265                if (obj instanceof Byte) {
266                        gen.write(((Byte)obj).intValue());
267                        return;
268                } 
269                if (obj instanceof Number) {
270                        if (obj instanceof Short) {
271                                gen.write((Short)obj);
272                                return;
273                        }
274                        if (obj instanceof Integer) {
275                                gen.write((Integer)obj);
276                                return;
277                        }
278                        if (obj instanceof Long) {
279                                gen.write((Long)obj);
280                                return;
281                        }
282                        if (obj instanceof BigInteger) {
283                                gen.write((BigInteger)obj);
284                                return;
285                        }
286                        if (obj instanceof BigDecimal) {
287                                gen.write((BigDecimal)obj);
288                                return;
289                        }
290                        gen.write((Double)obj);
291                        return;
292                }
293                PropertyEditor propertyEditor = PropertyEditorManager
294                        .findEditor(obj.getClass());
295                if (propertyEditor != null) {
296                        propertyEditor.setValue(obj);
297                        gen.write(propertyEditor.getAsText());
298                        return;
299                }
300                if (obj instanceof Object[]) {
301                        gen.writeStartArray();
302                        Class<?> compType = null;
303                        if (expectedType != null && expectedType.isArray()) {
304                                compType = expectedType.getComponentType();
305                        }
306                        for (Object item: (Object[])obj) {
307                                doWriteObject(item, compType);
308                        }
309                        gen.writeEnd();
310                        return;
311                }
312                if (obj instanceof Collection) {
313                        gen.writeStartArray();
314                        for (Object item: (Collection<?>)obj) {
315                                doWriteObject(item, null);
316                        }
317                        gen.writeEnd();
318                        return;
319                }
320                if (obj instanceof Map) {
321                        @SuppressWarnings("unchecked")
322                        Map<String,Object> map = (Map<String,Object>)obj;
323                        gen.writeStartObject();
324                        for (Map.Entry<String, Object> e: map.entrySet()) {
325                                gen.writeKey(e.getKey());
326                                doWriteObject(e.getValue(), null);
327                        }
328                        gen.writeEnd();
329                        return;
330                }
331                try {
332                        BeanInfo beanInfo = Introspector.getBeanInfo(
333                                        obj.getClass(), Object.class);
334                        if (beanInfo.getPropertyDescriptors().length > 0) {
335                                gen.writeStartObject();
336                                if (!obj.getClass().equals(expectedType)) {
337                                        gen.write("class", aliases.computeIfAbsent(
338                                                        obj.getClass(), k -> k.getName()));
339                                }
340                                for (PropertyDescriptor propDesc: 
341                                                beanInfo.getPropertyDescriptors()) {
342                                        if (propDesc.getValue("transient") != null) {
343                                                continue;
344                                        }
345                                        if (excluded.contains(propDesc.getPropertyType().getName())) {
346                                                continue;
347                                        }
348                                        Method method = propDesc.getReadMethod();
349                                        if (method == null) {
350                                                continue;
351                                        }
352                                        try {
353                                                Object value = method.invoke(obj);
354                                                gen.writeKey(propDesc.getName());
355                                                doWriteObject(value, propDesc.getPropertyType());
356                                                continue;
357                                        } catch (IllegalAccessException | IllegalArgumentException
358                                                | InvocationTargetException e) {
359                                                // Bad luck
360                                        }
361                                }
362                                gen.writeEnd();
363                                return;
364                        }
365                } catch (IntrospectionException e) {
366                        // No luck
367                }
368                // Last resort
369                gen.write(obj.toString());
370        }
